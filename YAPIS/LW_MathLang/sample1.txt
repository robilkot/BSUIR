// Declaration above other stuff:
sub calculateHypotenuse(float a, float b, float result) {
	result = (a^2 + b^2)^0.5
}

sub isRightTriangle(float a, float b, float c, bool result) {
	result = (a^2 + b^2 == c^2) or (a^2 + c^2 == b^2) or (b^2 + c^2 == a^2)
}

sub usingGlobalStuff() {
	global float global_var1
	global_var1 = 3.0
	
	float global_var2
	global_var2 = 3.0
	
	write(global_var3) // 2.0
}

// Start of program
int counter = 0
float cathetusA, cathetusB = 0.0

write("Enter the first cathetus: ")
cathetusA = read()
write("Enter the second cathetus: ")
cathetusB = read()

// Compound assignment
float area, hypotenuse = 0.0
area, hypotenuse = (cathetusA * cathetusB / 2.0), calculateHypotenuse(cathetusA, cathetusB)

// Branching operator
if (hypotenuse > 10.0) {
	write("Large triangle. ")
} else {
	write("Small triangle. ")
}

write("Area: ", area, ", Hypotenuse: ", hypotenuse)

// Type conversion
int roundedHypotenuse = int(hypotenuse)
write("Rounded hypotenuse: ", roundedHypotenuse)

bool isRight = isRightTriangle(cathetusA, cathetusB, hypotenuse)
write("Is right triangle? ", isRight)

// until loop
int x = 0
until(x == 10) {
	write(x + 1)
}
